149    /**
150     *Procedure qui a pour but d'appeler la bonne methode en fonction de la commande passé en parametre
151     *@param pCommandLine un String contenant la commande tapée par le joueur
152    */
153    public void interpretCommand(String pCommandLine) 
154    {   
155        if(time())return;
156        this.aGui.println(pCommandLine);
157        Command command = this.aParser.getCommand(pCommandLine);
158
159        if(command.isUnknown()) {
160            this.aGui.println("I don't know what you mean...");
161            return;
162        }
163
164        CommandWord commandWord = command.getCommandWord();
165        switch(commandWord){
166            case HELP  : printHelp(); break;
167            case GO    : goRoom(command); break;
168            case TEST  : test(command); break;
169            case TAKE  : take(command); break;
170            case DROP  : drop(command); break;
171            case LOOK  : look(); break;
172            case EAT   : eat(command); break;
173            case BACK  : back(); break;
174            case ITEMS : items(); break;
175            case CHARGE: charge(command); break;
176            case TP    : teleport(command); break;
177            case QUIT  : if(command.hasSecondWord())
178                             this.aGui.println("Quit what?");
179                         else
180                             endGame();
181                         break;
182        }
183    }

197    /**
198     *procedure effectuee lorsque la commande go est tapee
199     *@param pCommand le deuxieme String de la commande tapee par le joueur qui est la direction
200    */
201    private void goRoom(final Command pCommand)
202    {
203        if(!pCommand.hasSecondWord())
204        {
205            this.aGui.println("go where ?");
206            return;
207        }  
208        
209        String vDirection = pCommand.getSecondWord();
210
211        Room vNextRoom = this.aPlayer.getLocalisation().getExit(vDirection);
212
213        if (vNextRoom == null) this.aGui.println("There is no door !");
214        else
215        {
216            if(this.aPlayer.getLocalisation().isTrapDoor(vDirection)==true) this.aPlayer.resetLastRoom();
217            else this.aPlayer.setLastRoom(this.aPlayer.getLocalisation());
218            this.aPlayer.setLocalisation(vNextRoom);
219            this.aGui.println(this.aPlayer.getLocalisation().getLongDescription());
220            if(this.aPlayer.getLocalisation().getImageName() != null)
221                this.aGui.showImage(this.aPlayer.getLocalisation().getImageName());
222        }
223    }

225    /**
226     *procedure effectuee lorsque la commande test est tapee
227     *@param pCommand le deuxieme String de la commande tapee par le joueur qui est l'emplacement du fichier conteant la suite des commande de test
228    */
229    private void test(final Command pCommand)
230    {
231        if(!pCommand.hasSecondWord())
232        {
233            this.aGui.println("test what?");
234            return;
235        }         
236
237        String vFile = pCommand.getSecondWord();
238        Scanner vScan = null;
239
240        try {vScan = new Scanner(new File("./"+vFile+".txt"));}
241        catch ( final java.io.FileNotFoundException pException )
242        {
243            this.aGui.println("File not find");
244        }
245
246        while(vScan.hasNextLine())
247        {
248            String vLigne = vScan.nextLine();
249            interpretCommand(vLigne);
250        }
251        if (vScan != null) {vScan.close();}
252    }

273    /**
274     *procedure efectuee lorsque la commande eat est tapee
275     *@param pCommand le deuxieme String de la commande tapee par le joueur qui est l'item a manger
276    */
277    private void eat(final Command pCommand)
278    {
279        if(!pCommand.hasSecondWord())
280        {
281            this.aGui.println("eat what?");
282            return;
283        }   
284
285        String vItem = pCommand.getSecondWord();
286
287        Item vToEat = this.aPlayer.getItem(vItem);
288
289        if(vToEat == null) this.aGui.println("I don't have it !");
290        else if(vItem.equals("magicCookie")){
291            this.aPlayer.takeItem(vItem, vToEat);
292            this.aPlayer.getLocalisation().dropItem(vItem);
293            this.aPlayer.setStrong(this.aPlayer.getStrong()+100);
294            this.aGui.println("You have eat the magic cookie");
295        }
296        else
297            this.aGui.println("You have eaten now and you are not hungry any more.");
298    }

344        }
345    }

429    /**
430     *methode qi veririfie si le nombre de commande max rentrer n'est pas depacee
431     *@return true si le nombre de commande est depacee
432    */
433    private boolean time(){
434        this.aCommantInput++;
435        if(this.aCommantInput>200){
436            this.aGui.println("You are out of time");
437            this.aGui.enable(false);
438            return true;
439        }
440        else return false;
441    }
