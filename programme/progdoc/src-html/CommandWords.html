001/**
002 * This class is part of the "World of Zuul" application. 
003 * "World of Zuul" is a very simple, text based adventure game.  
004 * 
005 * This class holds an enumeration table of all command words known to the game.
006 * It is used to recognise commands as they are typed in.
007 *
008 * @author  Espinasse Baptiste
009 * @version 2017.12.16
010 */
011
012import java.util.HashMap;
013
014public class CommandWords
015{
016    private HashMap<String,CommandWord> aValidCommands;
017
018    /**
019     * Constructeur par defaut
020     */
021    public CommandWords()
022    {
023        this.aValidCommands = new HashMap<String, CommandWord>();
024        for(CommandWord command : CommandWord.values()){
025            if(command != CommandWord.UNKNOWN)
026                this.aValidCommands.put(command.toString(), command);
027        }
028    } // CommandWords()
029    /**
030    *Cette methode a pour but de retouver la commande souhaiter dans le hasmap
031    *@param pCommandWord Un String correspondant a la commande que l'on souhaite realiser
032    *@return Le CommandWord correspondant au String en parametre
033    */
034    public CommandWord getCommandWord(String pCommandWord)
035    {
036        CommandWord vCommand = this.aValidCommands.get(pCommandWord);
037        if(vCommand != null) {
038            return vCommand;
039        }
040        else {
041            return CommandWord.UNKNOWN;
042        }
043    }
044
045    /**
046     * Verifie si une String donnee fait partie des commandes valides. 
047     * @param pString la String a tester
048     * @return true si pString est une comande valide, false sinon
049     */
050    public boolean isCommand( final String pString )
051    {
052        return this.aValidCommands.containsKey(pString);
053    } // isCommand()
054
055    /**
056     * cette method Ã  pour but de retournee un String contenant toute les commandes realisable
057     *@return Un String contenant toute les commandes
058     */
059    public String getCommandList() 
060    {
061        StringBuilder commands = new StringBuilder();
062        for(String vCommand : this.aValidCommands.keySet()) {
063            commands.append( vCommand + "  " );
064        }
065        return commands.toString(); 
066    }
067
068} // CommandWords
