141    /**
142     *Procedure qui a pour but d'appeler la bonne methode en fonction de la commande passé en parametre
143     *@param pCommandLine un String contenant la commande tapée par le joueur
144    */
145    public void interpretCommand(String pCommandLine) 
146    {
147        this.aGui.println(pCommandLine);
148        Command command = this.aParser.getCommand(pCommandLine);
149
150        if(command.isUnknown()) {
151            this.aGui.println("I don't know what you mean...");
152            return;
153        }
154
155        String commandWord = command.getCommandWord();
156        if (commandWord.equals("help"))        printHelp();
157        else if (commandWord.equals("go"))     goRoom(command);
158        else if (commandWord.equals("look"))   look();
159        else if (commandWord.equals("eat"))    eat();
160        else if (commandWord.equals("back"))   back();
161        else if (commandWord.equals("quit")) {
162            if(command.hasSecondWord())
163                this.aGui.println("Quit what?");
164            else
165                endGame();
166        }
167    }

181    /**
182     *procedure effectuee lorsque la commande go est tapee
183     *@param pCommand le deuxieme String de la commande tapee par le joueur qui est la direction
184    */
185    private void goRoom(final Command pCommand)
186    {
187        if(!pCommand.hasSecondWord())
188        {
189            this.aGui.println("go where ?");
190            return;
191        }  
192        
193        String vDirection = pCommand.getSecondWord();
194
195        Room vNextRoom = this.aCurrentRoom.getExit(vDirection);
196
197        if (vNextRoom == null) this.aGui.println("There is no door !");
198        else
199        {
200            this.aLastRooms.push(this.aCurrentRoom);
201            this.aCurrentRoom = vNextRoom;
202            this.aGui.println(this.aCurrentRoom.getLongDescription());
203            if(this.aCurrentRoom.getImageName() != null)
204                this.aGui.showImage(this.aCurrentRoom.getImageName());
205        }
206    }
