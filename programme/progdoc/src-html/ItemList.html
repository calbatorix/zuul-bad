001import java.util.HashMap;
002import java.util.Set;
003/**
004 * This class is part of the "World of Zuul" application. 
005 * "World of Zuul" is a very simple, text based adventure game.  
006 * 
007 * @author  Espinasse Baptiste
008 * @version 2017.12.16
009 */
010public class ItemList
011{
012    private HashMap<String, Item> aInventary;
013
014    /**
015     * Constructeur d'objets de classe ItemList
016     */
017    public ItemList()
018    {
019        this.aInventary = new HashMap<>();
020    }
021
022    /**
023     *cette procedure a pour but d'afficher tout les items contenu dans l'inventaire dela piece ou du player
024     *@return Un String de tout les items contenu dans l'inventaire
025    */
026    public String getItemString(){
027        String vReturnString = "";
028        Set<String> vKeys = this.aInventary.keySet();
029        for(String vItem : vKeys)
030        {
031            vReturnString += " a "+vItem+"\n";
032        }
033        return vReturnString;
034    }
035
036    /**
037     *methode permettant de mettre un item en plus dans l'inventaire
038     *@param pStringItem est un String correspondant a la description/nom de l'item a ajouter dans l'inventaire
039     *@param pItem est l'item a ajouter dansl'inventaire
040    */
041    public void takeItem(final String pStringItem, final Item pItem){this.aInventary.put(pStringItem, pItem);}
042
043    /**
044     *methode permettant de retirer un item de l'inventaire
045     *@param pStringItem Un string contenant la description/nom de l'item a retirer de l'inventaire
046    */
047    public void dropItem(final String pStringItem){this.aInventary.remove(pStringItem);}
048
049    /**
050     *getter d'un item a partir de la description/nom de l'item
051     *@param pItem est un String correspondant a la description/nom de l'item que l'on souhaite obtenir dans l'invantaire
052    */
053    public Item getItem(final String pItem){return this.aInventary.get(pItem);}
054
055}
