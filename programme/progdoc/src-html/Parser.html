001import java.util.StringTokenizer;
002/**
003 * This class is part of the "World of Zuul" application. 
004 * "World of Zuul" is a very simple, text based adventure game.  
005 * 
006 * This parser reads user input and tries to interpret it as an "Adventure"
007 * command. Every time it is called it reads a line from the terminal and
008 * tries to interpret the line as a two word command. It returns the command
009 * as an object of class Command.
010 *
011 * The parser has a set of known command words. It checks user input against
012 * the known commands, and if the input is not one of the known commands, it
013 * returns a command object that is marked as an unknown command.
014 * 
015 * @author  Espinasse Baptiste
016 * @version 2017.12.16
017 */
018public class Parser 
019{
020    private CommandWords aValidCommands;  // (voir la classe CommandWords)
021
022    /**
023     * Constructeur par defaut qui cree les 2 objets prevus pour les attributs
024     */
025    public Parser() 
026    {
027        this.aValidCommands = new CommandWords();
028    } // Parser()
029
030    /**
031     * Methode ayant pour but manufacturer les commamande rentrer par l'utilisateur pour qu'il soit compreansible par les autres methode
032     *@param pInputLine est le String renter par l'utilisateur dans la l'interface IHM
033     *@return The next command from the user.
034     */
035    public Command getCommand(final String pInputLine) 
036    {
037        String vWord1 = null;
038        String vWord2 = null;
039
040        StringTokenizer vTokenizer = new StringTokenizer( pInputLine );
041
042        if ( vTokenizer.hasMoreTokens() )
043            vWord1 = vTokenizer.nextToken();      // get first word
044        else
045            vWord1 = null;
046
047        if ( vTokenizer.hasMoreTokens() )
048            vWord2 = vTokenizer.nextToken();      // get second word
049        else
050            vWord2 = null;
051
052        if ( this.aValidCommands.isCommand( vWord1 ) )
053            return new Command( this.aValidCommands.getCommandWord(vWord1), vWord2 );
054        else
055            return new Command( null, vWord2 );
056    } // getCommand()
057
058    /**
059     *
060     *
061    */
062    public String showCommands()
063    {
064        return this.aValidCommands.getCommandList();
065    }
066} // Parser
