001import javax.swing.JFrame;
002import javax.swing.JTextField;
003import javax.swing.JTextArea;
004import javax.swing.JLabel;
005import javax.swing.JPanel;
006import javax.swing.JButton;
007import javax.swing.ImageIcon;
008import javax.swing.JScrollPane;
009import java.awt.BorderLayout;
010import java.awt.Dimension;
011import java.awt.GridLayout;
012import java.awt.event.ActionListener;
013import java.awt.event.ActionEvent;
014import java.awt.event.WindowAdapter;
015import java.awt.event.WindowEvent;
016import java.net.URL;
017/**
018 * This class implements a simple graphical user interface with a text entry
019 * area, a text output area and an optional image.
020 * 
021 * @author  Espinasse Baptiste
022 * @version 2017.12.16
023 */
024public class UserInterface implements ActionListener
025{
026    private GameEngine aEngine;
027    private JFrame     aMyFrame;
028    private JTextField aEntryField;
029    private JTextArea  aLog;
030    private JLabel     aImage;
031    private JPanel     aButton;
032    private JButton    aButtonN, aButtonS ,aButtonE, aButtonW, aButtonU, aButtonD;
033    private JButton    aButtonEat, aButtonLook, aButtonHelp;
034    /**
035     * Construct a UserInterface. As a parameter, a Game Engine
036     * (an object processing and executing the game commands) is
037     * needed.
038     * 
039     * @param pGameEngine  The GameEngine object implementing the game logic.
040     */
041    public UserInterface( final GameEngine pGameEngine )
042    {
043        this.aEngine = pGameEngine;
044        this.createGUI();
045    } // UserInterface(.)
046
047    /**
048     *@param pText un String de ce u il faut afficher
049     * Print out some text into the text area.
050     */
051    public void print( final String pText )
052    {
053        this.aLog.append( pText );
054        this.aLog.setCaretPosition( this.aLog.getDocument().getLength() );
055    } // print(.)
056
057    /**
058       *@param pText un String de ce u il faut afficher
059     * Print out some text into the text area, followed by a line break.
060     */
061    public void println( final String pText )
062    {
063        this.print( pText + "\n" );
064    } // println(.)
065
066    /**
067     * Show an image file in the interface.
068     */
069    public void showImage( final String pImageName )
070    {
071        URL vImageURL = this.getClass().getClassLoader().getResource( pImageName );
072        if ( vImageURL == null )
073            System.out.println( "image not found" );
074        else {
075            ImageIcon vIcon = new ImageIcon( vImageURL );
076            this.aImage.setIcon( vIcon );
077            this.aMyFrame.pack();
078        }
079    } // showImage(.)
080
081    /**
082     * Enable or disable input in the input field.
083     */
084    public void enable( final boolean pOnOff )
085    {
086        this.aEntryField.setEditable( pOnOff );
087        if ( ! pOnOff )
088            this.aEntryField.getCaret().setBlinkRate( 0 );
089
090    } // enable(.)
091
092    /**
093     * Set up graphical user interface.
094     */
095    private void createGUI()
096    {
097        this.aMyFrame = new JFrame( "Zork" );
098        this.aEntryField = new JTextField( 34 );
099
100        this.aLog = new JTextArea();
101        this.aLog.setEditable( false );
102        JScrollPane vListScroller = new JScrollPane( this.aLog );
103        vListScroller.setPreferredSize( new Dimension(200, 200) );
104        vListScroller.setMinimumSize( new Dimension(100,100) );
105
106        JPanel vPanel = new JPanel();
107        this.aImage = new JLabel();
108        this.makeBoutonBar();
109        
110        //this.aButton1.addActionListener();
111        
112        vPanel.setLayout( new BorderLayout() );
113        vPanel.add( this.aImage, BorderLayout.NORTH );
114        vPanel.add( vListScroller, BorderLayout.CENTER );
115        vPanel.add( this.aEntryField, BorderLayout.SOUTH );
116        vPanel.add( this.aButton, BorderLayout.EAST );
117
118        this.aMyFrame.getContentPane().add( vPanel, BorderLayout.CENTER );
119
120        // add some event listeners to some components
121        this.aMyFrame.addWindowListener( new WindowAdapter() {
122            public void windowClosing(WindowEvent e) { System.exit(0); }
123        } );
124
125        this.aEntryField.addActionListener( this );
126
127        this.aMyFrame.pack();
128        this.aMyFrame.setVisible( true );
129        this.aEntryField.requestFocus();
130    } // createGUI()
131
132    public void makeBoutonBar()
133    {
134        aButton = new JPanel();
135        aButton.setLayout(new GridLayout(0,1,3,5));
136        
137        this.aButtonN = new JButton("north");        
138        this.aButtonN.addActionListener(this);
139        this.aButtonS = new JButton("south");
140        this.aButtonS.addActionListener(this);
141        this.aButtonE = new JButton("east");
142        this.aButtonE.addActionListener(this);
143        this.aButtonW = new JButton("west");
144        this.aButtonW.addActionListener(this);
145        this.aButtonEat = new JButton("eat"); 
146        this.aButtonEat.addActionListener(this);
147        this.aButtonLook = new JButton("look"); 
148        this.aButtonLook.addActionListener(this);
149        this.aButtonHelp = new JButton("help"); 
150        this.aButtonHelp.addActionListener(this);
151
152        aButton.add( this.aButtonN);
153        aButton.add( this.aButtonS);
154        aButton.add( this.aButtonE);
155        aButton.add( this.aButtonW);
156        aButton.add( this.aButtonEat);
157        aButton.add( this.aButtonHelp);
158        aButton.add( this.aButtonLook);
159
160    }
161    
162    /**
163     * Actionlistener interface for entry textfield.
164     */
165    public void actionPerformed( final ActionEvent pE ) 
166    {
167        // no need to check the type of action at the moment.
168        // there is only one possible action: text entry
169        if(pE.getSource() == this.aButtonN){ this.aEngine.interpretCommand("go north");}
170        if(pE.getSource() == this.aButtonS){ this.aEngine.interpretCommand("go south");}
171        if(pE.getSource() == this.aButtonE){ this.aEngine.interpretCommand("go east");}
172        if(pE.getSource() == this.aButtonW){ this.aEngine.interpretCommand("go west");}
173        if(pE.getSource() == this.aButtonEat){ this.aEngine.interpretCommand("eat");}
174        if(pE.getSource() == this.aButtonLook){ this.aEngine.interpretCommand("look");}
175        if(pE.getSource() == this.aButtonHelp){ this.aEngine.interpretCommand("help");}
176        if(pE.getSource() == this.aEntryField) {processCommand();}
177    } // actionPerformed(.)
178
179    /**
180     * A command has been entered. Read the command and do whatever is 
181     * necessary to process it.
182     */
183    private void processCommand()
184    {
185        String vInput = this.aEntryField.getText();
186        this.aEntryField.setText( "" );
187
188        this.aEngine.interpretCommand( vInput );
189    } // processCommand()
190} // UserInterface
